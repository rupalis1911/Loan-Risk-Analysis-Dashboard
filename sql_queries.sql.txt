-- labeling as invalid, normal or suspicious = data validation

ALTER TABLE loanapproval
ADD COLUMN age_validation varchar(20);
UPDATE loanapproval
SET age_validation =
	CASE 
		WHEN age<18 THEN 'Invalid'
        WHEN age BETWEEN 18 AND 65 THEN 'Normal'
		ELSE 'Suspicious'
        END;

-- age segmentation
ALTER TABLE loanapproval
ADD COLUMN applicant_category varchar(20);
UPDATE loanapproval
SET applicant_category =
	CASE 
		WHEN age<18 then 'Invalid'
        WHEN age BETWEEN 18 AND 25 then 'Early Career'
		WHEN age BETWEEN 26 AND 35 then 'Young Professionals'
		WHEN age BETWEEN 36 AND 45 then 'Mid Career'
        WHEN age BETWEEN 46 AND 60 then 'Experienced'
        ELSE 'Senior Applicant'
        END;

		WHEN credit_score BETWEEN 670 AND 739 then 'Good'
        WHEN credit_score BETWEEN 740 AND 799 then 'Very Good'
        WHEN credit_score BETWEEN 800 AND 850 then 'Excellent'
        END;

UPDATE loanapproval
SET affordability_band =
	CASE 
		WHEN affordability_ratio> 0.25 THEN 'Low Exposure'
		WHEN affordability_ratio BETWEEN 0.25 AND 0.50 THEN 'Moderate Exposure'
		WHEN affordability_ratio BETWEEN 0.50 AND 0.1 THEN 'High Exposure'
		WHEN affordability_ratio BETWEEN 1.00 AND 1.50 THEN 'Very High Exposure'
        ELSE 'Extreme Exposure'
        END;

ALTER table loanapproval
ADD COLUMN employment_stability_group varchar(20);
UPDATE loanapproval
SET employment_stability_group =
		CASE
            WHEN employment_status = 'unemployed' then 'Income Risk'
			WHEN employment_status = 'employed' then 'Stable Income'
            WHEN employment_status = 'Self-employed' then 'Variable Income'
            END;
            
SELECT DISTINCT credit_score_band, affordability_band, employment_stability_group FROM
loanapproval;

ALTER TABLE loanapproval	
ADD COLUMN credit_score_points	int;
UPDATE loanapproval
SET credit_score_points =
	CASE 
		WHEN credit_score_band = 'Poor' THEN 4
        WHEN credit_score_band = 'Fair' THEN 3
		WHEN credit_score_band = 'Good' THEN 2
        WHEN credit_score_band = 'Very Good' THEN 1
        WHEN credit_score_band = 'Excellent' THEN 0
        ELSE 'NULL'
	END;
    
ALTER TABLE loanapproval	
ADD COLUMN exposure_points	int;
UPDATE loanapproval
SET exposure_points =
	CASE 
		WHEN affordability_band = 'Extreme Exposure' THEN 2
		WHEN affordability_band = 'Moderate Exposure' THEN 1
        WHEN affordability_band = 'Low Exposure' THEN 0
        Else 'NULL'
	END;
    
ALTER TABLE loanapproval	
ADD COLUMN employment_points int;
UPDATE loanapproval
SET employment_points =
	CASE 
		WHEN employment_stability_group = 'Income Risk' THEN 2
		WHEN employment_stability_group = 'Variable Income' THEN 1
        WHEN employment_stability_group = 'Stable Income' THEN 0
        Else 'NULL'
	END;
	
ALTER table loanapproval
ADD COLUMN total_risk_score INT;
UPDATE loanapproval
SET total_risk_score = credit_score_points + exposure_points + employment_points;

SELECT total_risk_score, COUNT(*)
FROM loanapproval
GROUP BY total_risk_score
ORDER BY total_risk_score DESC;

        
ALTER table loanapproval

	CASE
		WHEN total_risk_score BETWEEN 0 AND 2 THEN 'Low Risk'
        WHEN total_risk_score BETWEEN 3 AND 4 THEN 'Moderate Risk'
        WHEN total_risk_score BETWEEN 5 AND 6 THEN 'High Risk'
        WHEN total_risk_score BETWEEN 7 AND 8 THEN 'Critical Risk'
        Else Null
        END;

-- Approval Rate by Risk Category
SELECT risk_category, COUNT(*) AS total_applications,
sum(loan_approved) AS approved_count,
ROUND(AVG(loan_approved) * 100, 2) AS approval_rate_percent
FROM loanapproval
GROUP BY risk_category
ORDER BY approval_rate_percent DESC;
